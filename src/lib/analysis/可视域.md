#  可视域

<div data-sunwayTheme='light'>
  <EpsplanetEarth :showDefaultBasemap="true" @onReady="ready">
    <EpsplanetButton container="#earthContainer" icon="icon-tool_kjcx" type="panel" :position="position" :panel="panel" title='可视域分析'>
      <EpsplanetViewshed/>
    </EpsplanetButton>
  </EpsplanetEarth>
</div>

<script setup lang='ts'>
import {ref} from 'vue';
const position = ref({
  left: 10,
  top: 10
});
const panel = ref({
  position: {
    left: 10,
    top: 10
  }
});
const ready = (earth:any)=>{
  earth.sceneTree.root.children.push(
  {
    ref: 'tileset',
    czmObject: {
      "xbsjType": "Tileset",
      "xbsjGuid": "d3266895-4795-41a1-92f3-46be5edc6532",
      "name": "大雁塔",
      "url": "/EPSGIS-DEV-PORTAL/Demos/last/assets/dayanta/tileset.json",
    }
  }
  )
  earth.sceneTree.$refs.tileset.czmObject.flyTo()
}
</script>
